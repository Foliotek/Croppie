Croppie

Croppie is a fast, easy to use image cropping plugin with tons of configuration options!

Usage


$("#item").imageCropper({
    viewport: {
        width: px,
        height: px,
        type: 'square|circle' //default 'square'
    },
    boundary: {
        width: px,
        height: px
    },
    customClass: '',
    showZoom: true|false, //default true
    update: function (cropper) { }
});

// Bind an image to the cropper
$('#item').imageCropper('bind', 'path/to/image.png');

// Get the crop data from a cropper
var cropData = $('#item').imageCropper('get');
